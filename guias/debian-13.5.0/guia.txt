# NO PRIMEIRO BOOT, É COMUM O RFKILL BLOQUEAR PLACA DE REDE (E ALGUNS OUTROS DISPOSITIVOS). NESSE CASO, O MÉTODO MAIS SIMPLES PARA RESOLVER É LIGAR A ANCORAGEM USB DO CELULAR PARA CONECTAR O COMPUTADOR AO WIFI, INSTALAR O RFKILL, DESBLOQUEAR A PLACA DE REDE, INSTALAR O NETWORK-MANAGER E ENTÃO UTILIZAR O NMTUI PARA CONECTAR AO WIFI.
RFKILL: https://docs.kernel.org/driver-api/rfkill.html
NETWORK-MANAGER: https://networkmanager.dev/docs/api/latest/
	su -
	apt update
	apt install rfkill network-manager
	rfkill list
	rfkill unblock all
	systemctl enable --now NetworkManager
	nmtui
	reboot

# COMO A INSTALAÇÃO MÍNIMA NÃO VEM COM O PACOTE SUDO, SEGUE O PROCESSO PARA INSTALAR E DAR PERMISSÕES SUDO AO USUÁRIO
SUDO: https://www.sudo.ws/docs/man/1.8.14/sudo.man/
	su -
	apt update
	apt install sudo
	usermod -aG sudo nome_usuario
	su - nome_usuario

# PARA FAZER O TAP-TO-CLICK DO TOUCHPAD FUNCIONAR E CONFIGURAR A ROLAGEM NATURAL, CRIE O ARQUIVO /etc/X11/xorg.conf.d/40-touchpad.conf E ADICIONE O CÓDIGO ABAIXO
	Section "InputClass"
		Identifier "libinput touchpad catchall"
		MatchIsTouchpad "on"
		MatchDevicePath "/dev/input/event*"
		Driver "libinput"
		Option "Tapping" "on"
		Option "NaturalScrolling" "true"
	EndSection
# DEPOIS, DÊ reboot OU sudo systemctl restart display-manager
