1. NO PRIMEIRO BOOT, É COMUM O RFKILL BLOQUEAR PLACA DE REDE (E ALGUNS OUTROS DISPOSITIVOS). NESSE CASO, O MÉTODO MAIS SIMPLES PARA RESOLVER É LIGAR A ANCORAGEM USB DO CELULAR PARA CONECTAR O COMPUTADOR AO WIFI, INSTALAR O RFKILL, DESBLOQUEAR A PLACA DE REDE, INSTALAR O NETWORK-MANAGER E ENTÃO UTILIZAR O NMTUI PARA CONECTAR AO WIFI.
	su -
	apt update
	apt install rfkill network-manager
	rfkill list
	rfkill unblock all
	systemctl enable --now NetworkManager
	nmtui
	reboot

2. COMO A INSTALAÇÃO MÍNIMA NÃO VEM COM O PACOTE SUDO, SEGUE O PROCESSO PARA INSTALAR E DAR PERMISSÕES SUDO AO USUÁRIO
	su -
	apt update
	apt install sudo
	usermod -aG sudo nome_usuario
	su - nome_usuario

3. SEGUE ALGUNS FIRMWARES IMPORTANTES QUE TALVEZ NÃO VENHAM COM A INSTALAÇÃO E QUE PODEM EVITAR FUTURAS DORES DE CABEÇA
	sudo apt install linux-image-amd64 firmware-iwlwifi firmware-linux firmware-linux-nonfree

4. INSTALANDO O SERVIDOR GRÁFICO E O GERENCIADOR DE JANELAS
	sudo apt install xorg lightdm lightdm-gtk-greeter i3
	sudo systemctl enable --now lightdm

5. ESSES PACOTES SÃO PRÉ-REQUISITOS DO FLATPAK, QUE SERÁ O GERENCIADOR DE PACOTES QUE USAREMOS (ALÉM DO APT)
	sudo apt install ca-certificates apt-transport-https

6. INSTALANDO O NAVEGADOR (FIREFOX) VIA GERENCIADOR DE PACOTES FLATPAK
	sudo apt install flatpak
	sudo flatpak remote-add --if-not-exists flathub https://dl.flathub.org/repo/flathub.flatpakrepo
	sudo flatpak install flathub org.mozilla.firefox

7. PARA FAZER O ÁUDIO FUNCIONAR
	sudo apt install pipewire pipewire-pulse pipewire-audio-client-libraries wireplumber alsa-utils
	systemctl --user --now enable wireplumber
	wpctl status
	wpctl set-default ID
	alsamixer

8. PARA AUTENTICAÇÃO GRÁFICA (ALGUMAS FERRAMENTAS PRECISAM, COMO O VIRT-MANAGER)
    sudo apt install lxpolkit
ALÉM DISSO, ADICIONE AO FINAL DE ~/.config/i3/config A SEGUINTE LINHA
    exec --no-startup-id lxpolkit
E ENTÃO DÊ reboot

9. PARA UTILIZAR A VIRTUALIZAÇÃO DO KERNEL (KVM), AS BIBLIOTECAS ABAIXO SÃO NECESSÁRIAS
	sudo apt install qemu-kvm libvirt-daemon-system libvirt-clients virt-manager
	virt-manager

10. PARA FAZER O TAP-TO-CLICK DO TOUCHPAD FUNCIONAR E CONFIGURAR A ROLAGEM NATURAL, CRIE O ARQUIVO /etc/X11/xorg.conf.d/40-touchpad.conf E ADICIONE O CÓDIGO ABAIXO
	Section "InputClass"
		Identifier "libinput touchpad catchall"
		MatchIsTouchpad "on"
		MatchDevicePath "/dev/input/event*"
		Driver "libinput"
		Option "Tapping" "on"
		Option "NaturalScrolling" "true"
	EndSection
DEPOIS, DÊ reboot OU sudo systemctl restart display-manager

11. PARA INSTALAR O ROFI (GERENCIADOR DE APLICATIVOS)
	sudo apt install rofi
APÓS ISSO, VÁ ATÉ ~/.config/i3/config, COMENTE A LINHA bindsym $mod+d exec dmenu_run E ADICIONE ABAIXO A LINHA
	bindsym $mod+d exec --no-startup-id rofi -show drun

12. PARA INSTALAR O GERENCIADOR DE ARQUIVOS THUNAR
	sudo apt install thunar
